    会捷通 EVSDK(for VE260 新增部分) 开发文档

    API 参考

    getSerialNumber

    /**
  获取序列号，主要用于需要显示 SN 的地方使用
*/
String getSerialNumber();

    setProvision

    /**
  初始化 SDK 之后调用此方法，完成向平台注册
  server:服务器地址
  port:端口号
*/
int setProvision(String server, int port);

    注意，VE 260 等硬件终端需要使用 setProvision 接口向平台注册，而不是使用用户 login 的接口

    硬件终端使用前，需要先在平台自服务系统中添加这个终端 (在自服务系统 > 通讯录 > 终端列表 界面下，选择添加终端。添加终端时，需要终端的 SN 信息。)

    clearProvision

    /**
  取消注册。通常只需要在用户将终端从一个平台迁移到另一个平台，改变平台地址时使用该接口。
*/
int clearProvision();

    onRegister

    /**
  注册成功的回调，通过该通知获得注册状态的变化
  registered：
  		true :注册成功
  		false:注册失败
*/
void onRegister(boolean registered) {
  // do something, like displaying the status on UI
}

    getHardwareVersion

    /**
  获取硬件版本，主要用于显示
*/
String getHardwareVersion();

    enableFilterByName

    注意，该方法在 VE260 设备中必须使用，否则不会使用硬件编解码

    /**
  启用或关闭编解码器，
  	filterName：编解码器名字
  	enable：true 开启 / false 关闭
*/
int enableFilterByName(String filterName,boolean enable);

//示例代码
//初始化 SDK 时需要设置，对于 VE260 要关闭软解码器
 engine.enableFilterByName("MSX264Enc", false);
 engine.enableFilterByName("MSH264Dec", false);

    setCallLogMaxSize

    /**
  设置呼叫历史记录的最大值，上限500.
*/
int setCallLogMaxSize(int size);

    setLayoutCapacity

    注意，该方法在 VE260 设备上必须使用，按照以下例子中的能力集合设置即可

    /**
 设置终端支持分屏的集合，对于 VE260 需要按以下设置，初始化时需要设置
 mode:GalleryMode
 	  SpeakerMode
*/
int setLayoutCapacity(SVEngine.LayoutMode mode, List<SVEngine.LayoutType> type);

//示例代码
//初始化 SDK 时设置
 SVEngine.LayoutType gallery_layout[] = {
            SVEngine.LayoutType.type1,
            SVEngine.LayoutType.type2H,
            SVEngine.LayoutType.type4,
            SVEngine.LayoutType.type9,};
 SVEngine.LayoutType speaker_layout[] = {
            SVEngine.LayoutType.type1,
            SVEngine.LayoutType.type2V,
            SVEngine.LayoutType.type3_1T_2B,
            SVEngine.LayoutType.type4_3T_1B,
            SVEngine.LayoutType.type5_1L_4R,
            SVEngine.LayoutType.type6,
            SVEngine.LayoutType.type8,
    };

engine.setLayoutCapacity(SVEngine.LayoutMode.GalleryMode,Arrays.asList(gallery_layout))
engine.setLayoutCapacity(SVEngine.LayoutMode.SpeakerMode,Arrays.asList(speaker_layout))

    getCallLog

    /**
 获取呼叫历史记录，参数如下：
   String id;
   EVCommon.EvCallType type;//呼叫类型：CallSvc
   EVCommon.EvCallDir dir：
   					CallOutGoing：去电
   					CallIncoming：来电
   EVCommon.EvCallStatus status;//呼叫是否成功
   String displayName;//名字
   String peer;//会议号
   long startTime;//开始时间
   duration;//持续时间
   isAudioOnly;//是否音频入会
*/
ArrayList<EVCommon.EVCallLog> getCallLog();

    removeCallLog

    /**
删除呼叫历史记录
*/
int removeCallLog(String id);

    setLocalVideoPreviewWindow

    /**
 设置本地视频预览窗口，非呼叫中、需要预览本地视频时设置
 注：需要在 enablePreview() 之后调用
*/
int setLocalVideoPreviewWindow(Object localVideoPreview);

    enablePreview

    /**
启用预览,设置近端本地视频时，请开启此方法
*/
int enablePreview(boolean enable);

    setLocalContentPreviewWindow

    /**
 双流本地预览.
 注：需要在 enableContentPreview() 之后调用
*/
int setLocalContentPreviewWindow(Object localContentPreview);

    enableContentPreview

    /**
 启用预览，双流本地预览时，需要开启此方法
*/
int enableContentPreview(boolean enable);

    getIoCpuTemp

    /**
  获取Cpu温度
*/
int getIoCpuTemp();

    getCpuUsage

    /**
   获取cpu使用率
*/
int getCpuUsage();

    getMemoryUsage

    /**
   获取内存使用率
*/
int getMemoryUsage();

    setPowerOff

    /**
   开关机:
     true：关机
     false：重启
*/
void setPowerOff(boolean bPowerOff);

    setCamMotion

    /**
   控制摄像头转动 
   action : 
   		start = 1 
   		continue = 2 
   		stop = 3  
*/
void setCamMotion(int action, int val);

    callPreset

    /**
   激活预设位
*/
void callPreset(int index);

    setPreset

    /**
   设置预设位：
   	 index:0-9
*/
void setPreset(int index);

    clearPreset

    /**
   清除预设位：
     index：表示预设点
*/
void clearPreset(int index);

    setUserAgent

    /**
  设置用户代理
  userAgent：名字
  version：版本号
*/
int setUserAgent(String userAgent,String version);

    onContent

    /**
双流：
    StreamDir：//新加入两个参数
            Plugin ：接入双流
            Plugout：断开双流
    boolean isExtern：投屏
*/
void onContent(EVCommon.ContentInfo info){};
